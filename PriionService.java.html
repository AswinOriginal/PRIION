42       @Override
43       public void onNotificationPosted(StatusBarNotification sbn) {
44           if (sbn.getPackageName().equals(getPackageName())) return;
45           if (!shouldAnalyze(sbn.getPackageName())) return;
46   
47           String text = sbn.getNotification().extras.getString("android.text");
48           String title = sbn.getNotification().extras.getString("android.title");
49   
50           if (text == null) return;
51   
52           String category = analyze(text);
53           Log.d(TAG, "üì© " + title + ": " + text + " -> " + category);
54   
55           cancelNotification(sbn.getKey());
56           repostAsGrouped(sbn, title, text, category);
57       }

59       private boolean shouldAnalyze(String pkg) {
60           return pkg.contains("whatsapp") || pkg.contains("android.gm") || pkg.contains("messaging") || pkg.contains("mms");
61       }

63       private void repostAsGrouped(StatusBarNotification sbn, String title, String text, String category) {
64           NotificationManager manager = (NotificationManager) getSystemService(Context.NOTIFICATION_SERVICE);
65   
66           String channelId;
67           String groupKey;
68           String groupTitle;
69           int priority;
70           int iconRes;
71           int color;
72   
73           // Configure Visuals
74           switch (category) {
75               case "critical":
76                   channelId = CHANNEL_CRITICAL;
77                   groupKey = GROUP_KEY_CRITICAL;
78                   groupTitle = "üö® CRITICAL UPDATES";
79                   priority = NotificationCompat.PRIORITY_MAX;
80                   iconRes = android.R.drawable.ic_dialog_alert;
81                   color = Color.RED; // <--- Make it RED!
82                   break;
83               case "important":
84                   channelId = CHANNEL_IMPORTANT;
85                   groupKey = GROUP_KEY_IMPORTANT;
86                   groupTitle = "‚ö†Ô∏è Important Updates";
87                   priority = NotificationCompat.PRIORITY_DEFAULT;
88                   iconRes = android.R.drawable.ic_dialog_info;
89                   color = Color.BLUE; // <--- Make it BLUE!
90                   break;
91               default:
92                   channelId = CHANNEL_SILENT;
93                   groupKey = GROUP_KEY_SILENT;
94                   groupTitle = "üí¨ Casual Messages";
95                   priority = NotificationCompat.PRIORITY_LOW;
96                   iconRes = android.R.drawable.ic_dialog_email;
97                   color = Color.GRAY;
98                   break;
99           }
100  
101          // Convert the system icon ID to a real Bitmap Image
102          Bitmap largeIcon = drawableToBitmap(iconRes);
103  
104          // A. Post the Child Notification
105          NotificationCompat.Builder builder = new NotificationCompat.Builder(this, channelId)
106                  .setContentTitle(title)
107                  .setContentText(text)
108                  .setSmallIcon(iconRes)       // Shows in Status Bar (Top Left)
109                  .setLargeIcon(largeIcon)     // <--- Shows BIG icon in the notification card
110                  .setColor(color)             // <--- Colors the small icon and header text
111                  .setPriority(priority)
112                  .setGroup(groupKey)
113                  .setAutoCancel(true)
114                  .setContentIntent(sbn.getNotification().contentIntent);
115  
116          if (category.equals("critical")) builder.setCategory(NotificationCompat.CATEGORY_ALARM);
117          if (category.equals("low")) builder.setSilent(true);
118  
119          manager.notify((int) System.currentTimeMillis(), builder.build());
120  
121          // B. Post the Summary Notification
122          Notification summary = new NotificationCompat.Builder(this, channelId)
123                  .setContentTitle(groupTitle)
124                  .setContentText("Check updates")
125                  .setSmallIcon(iconRes)
126                  .setColor(color)             // Color the summary too
127                  .setGroup(groupKey)
128                  .setGroupSummary(true)
129                  .setPriority(priority)
130                  .setAutoCancel(true)
131                  .build();
132  
133          int summaryId = 0;
134          if (category.equals("critical")) summaryId = 1;
135          if (category.equals("important")) summaryId = 2;
136          if (category.equals("low")) summaryId = 3;
137  
138          manager.notify(summaryId, summary);
139      }

141      // Helper: Turns a Resource ID (int) into a Picture (Bitmap)
142      private Bitmap drawableToBitmap(int drawableId) {
143          Drawable drawable = ContextCompat.getDrawable(this, drawableId);
144          if (drawable == null) return null;
145  
146          Bitmap bitmap = Bitmap.createBitmap(drawable.getIntrinsicWidth(), drawable.getIntrinsicHeight(), Bitmap.Config.ARGB_8888);
147          Canvas canvas = new Canvas(bitmap);
148          drawable.setBounds(0, 0, canvas.getWidth(), canvas.getHeight());
149  
150          // If the icon is white (default), tint it black/dark so it shows up on white backgrounds
151          drawable.setTint(Color.DKGRAY);
152          drawable.draw(canvas);
153          return bitmap;
154      }

156      private void createNotificationChannels() {
157          NotificationManager manager = getSystemService(NotificationManager.class);
158          NotificationChannel c1 = new NotificationChannel(CHANNEL_CRITICAL, "Critical Alerts", NotificationManager.IMPORTANCE_HIGH);
159          c1.enableVibration(true);
160          NotificationChannel c2 = new NotificationChannel(CHANNEL_IMPORTANT, "Work/Important", NotificationManager.IMPORTANCE_DEFAULT);
161          NotificationChannel c3 = new NotificationChannel(CHANNEL_SILENT, "Casual/Silent", NotificationManager.IMPORTANCE_LOW);
162          c3.setShowBadge(false);
163          manager.createNotificationChannel(c1);
164          manager.createNotificationChannel(c2);
165          manager.createNotificationChannel(c3);
166      }

168      private String analyze(String text) {
169          String lowText = text.toLowerCase();
170          for (String k : CRITICAL) if (lowText.contains(k)) return "critical";
171          for (String k : IMPORTANT) if (lowText.contains(k)) return "important";
172          for (String k : LOW) if (lowText.contains(k)) return "low";
173          return "normal";
174      }
